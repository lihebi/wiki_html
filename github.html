Branch

git branch # 列出所有分支
git branch xxx # 新建
git checkout [branch] # 切换
git push origin [branch]
git checkout -b [branch] # 新建并切换
git branch -d [branch] # 删除本地分支
git push origin --delete [branch] # 删除remote的分支
